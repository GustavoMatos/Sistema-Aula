## Notas Técnicas

**Schema de Validação:**
```typescript
import { z } from 'zod'

const leadFormSchema = z.object({
  name: z.string().min(2, 'Nome deve ter pelo menos 2 caracteres'),
  phone: z.string()
    .regex(/^\d{10,11}$/, 'Telefone inválido')
    .transform(val => val.replace(/\D/g, '')),
  email: z.string().email('Email inválido').optional().or(z.literal('')),
  company: z.string().optional(),
  source: z.string().optional(),
  potential_value: z.number().min(0).optional(),
  stage_id: z.string().uuid().optional(),
  tags: z.array(z.string()).default([]),
  notes: z.string().optional(),
})
```

**LeadFormModal.tsx:**
```typescript
export function LeadFormModal({ lead, open, onClose }: Props) {
  const isEditing = !!lead
  const { data: stages } = useStages()
  const createLead = useCreateLead()
  const updateLead = useUpdateLead()

  const form = useForm<LeadFormData>({
    resolver: zodResolver(leadFormSchema),
    defaultValues: lead || {
      name: '',
      phone: '',
      email: '',
      company: '',
      source: '',
      potential_value: undefined,
      stage_id: stages?.[0]?.id,
      tags: [],
      notes: '',
    },
  })

  const onSubmit = async (data: LeadFormData) => {
    try {
      if (isEditing) {
        await updateLead.mutateAsync({ id: lead.id, ...data })
        toast.success('Lead atualizado')
      } else {
        await createLead.mutateAsync(data)
        toast.success('Lead criado')
      }
      onClose()
    } catch (error) {
      toast.error('Erro ao salvar lead')
    }
  }

  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent className="max-w-lg">
        <DialogHeader>
          <DialogTitle>{isEditing ? 'Editar Lead' : 'Novo Lead'}</DialogTitle>
        </DialogHeader>

        <Form {...form}>
          <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
            {/* Form fields */}
          </form>
        </Form>
      </DialogContent>
    </Dialog>
  )
}
```

**PhoneInput.tsx:**
```typescript
export function PhoneInput({ value, onChange }: Props) {
  const formatPhone = (val: string) => {
    const digits = val.replace(/\D/g, '').slice(0, 11)
    if (digits.length <= 2) return digits
    if (digits.length <= 7) return `(${digits.slice(0, 2)}) ${digits.slice(2)}`
    return `(${digits.slice(0, 2)}) ${digits.slice(2, 7)}-${digits.slice(7)}`
  }

  return (
    <Input
      value={formatPhone(value)}
      onChange={(e) => onChange(e.target.value.replace(/\D/g, ''))}
      placeholder="(11) 99999-9999"
    />
  )
}
```
