## Notas Técnicas

**LeadHistoryService:**
```typescript
export class LeadHistoryService {
  async record(data: RecordHistoryDTO): Promise<void> {
    await supabase.from('lead_history').insert({
      lead_id: data.lead_id,
      action: data.action,
      from_stage_id: data.from_stage_id,
      to_stage_id: data.to_stage_id,
      metadata: data.metadata || {},
      performed_by: data.user_id,
    })
  }

  async getHistory(leadId: string): Promise<LeadHistory[]> {
    const { data } = await supabase
      .from('lead_history')
      .select(`
        *,
        from_stage:kanban_stages!from_stage_id(name, color),
        to_stage:kanban_stages!to_stage_id(name, color),
        performed_by:users!performed_by(id, full_name)
      `)
      .eq('lead_id', leadId)
      .order('created_at', { ascending: false })

    return data || []
  }
}
```

**Integração no LeadsService:**
```typescript
async moveStage(leadId: string, newStageId: string, userId: string) {
  const { data: lead } = await supabase
    .from('leads')
    .select('stage_id')
    .eq('id', leadId)
    .single()

  const oldStageId = lead?.stage_id

  // Update lead
  await supabase
    .from('leads')
    .update({ stage_id: newStageId })
    .eq('id', leadId)

  // Record history
  await this.historyService.record({
    lead_id: leadId,
    action: 'stage_change',
    from_stage_id: oldStageId,
    to_stage_id: newStageId,
    user_id: userId,
  })
}
```
