## Cálculo de Horário

```typescript
function calculateScheduledTime(
  enteredAt: Date,
  delayHours: number,
  startHour: number,
  endHour: number
): Date {
  let scheduled = addHours(enteredAt, delayHours)

  const hour = scheduled.getHours()
  if (hour < startHour) {
    scheduled = setHours(scheduled, startHour)
  } else if (hour >= endHour) {
    scheduled = setHours(addDays(scheduled, 1), startHour)
  }

  return scheduled
}
```
