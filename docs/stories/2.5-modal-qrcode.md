## Fluxo de Conexão

```
1. Usuário clica "Conectar WhatsApp"
2. Modal abre, pede nome da instância
3. Usuário digita nome e clica "Criar"
4. Sistema cria instância via API
5. Sistema busca QR Code
6. QR Code é exibido
7. Usuário escaneia com WhatsApp
8. Polling detecta status "connected"
9. Modal fecha, toast de sucesso
10. Lista de instâncias atualiza
```

**ConnectWhatsAppModal.tsx:**
```typescript
export function ConnectWhatsAppModal({ open, onClose }: Props) {
  const [step, setStep] = useState<'name' | 'qr'>('name')
  const [instanceName, setInstanceName] = useState('')
  const [instanceId, setInstanceId] = useState<string | null>(null)

  const createInstance = useCreateInstance()
  const { data: qrData, refetch: refetchQR } = useQuery({
    queryKey: ['qr-code', instanceId],
    queryFn: () => api.whatsapp.getQRCode(instanceId!),
    enabled: !!instanceId && step === 'qr',
    refetchInterval: 20000, // Refresh QR every 20s
  })

  const { data: statusData } = useQuery({
    queryKey: ['instance-status', instanceId],
    queryFn: () => api.whatsapp.getStatus(instanceId!),
    enabled: !!instanceId && step === 'qr',
    refetchInterval: 3000, // Poll every 3s
  })

  // Auto-close when connected
  useEffect(() => {
    if (statusData?.status === 'connected') {
      toast.success('WhatsApp conectado com sucesso!')
      onClose()
    }
  }, [statusData?.status])

  const handleCreate = async () => {
    const instance = await createInstance.mutateAsync(instanceName)
    setInstanceId(instance.id)
    setStep('qr')
  }

  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent>
        {step === 'name' && (
          // Input form
        )}
        {step === 'qr' && (
          <QRCodeDisplay
            qrCode={qrData?.qrCode}
            pairingCode={qrData?.pairingCode}
            status={statusData?.status}
          />
        )}
      </DialogContent>
    </Dialog>
  )
}
```

export function QRCodeDisplay({ qrCode, pairingCode, status }: Props) {
  return (
    <div className="flex flex-col items-center gap-4">
      {qrCode ? (
        <img src={qrCode} alt="QR Code" className="w-64 h-64" />
      ) : (
        <div className="w-64 h-64 bg-muted animate-pulse rounded" />
      )}

      <p className="text-sm text-muted-foreground">
        Escaneie com seu WhatsApp
      </p>

      {pairingCode && (
        <p className="text-sm">
          Ou use o código: <code className="font-mono">{pairingCode}</code>
        </p>
      )}

      <div className="flex items-center gap-2 text-sm text-muted-foreground">
        <Loader2 className="w-4 h-4 animate-spin" />
        Aguardando conexão...
      </div>
    </div>
  )
}
