// analytics.service.ts
export async function getOverviewMetrics(
  workspaceId: string,
  period: '7d' | '30d' | '90d'
): Promise<OverviewMetrics> {
  const days = parseInt(period)
  const periodStart = new Date()
  periodStart.setDate(periodStart.getDate() - days)

  // Total de leads
  const { count: totalLeads } = await supabase
    .from('leads')
    .select('*', { count: 'exact', head: true })
    .eq('workspace_id', workspaceId)

  // Leads no período
  const { count: leadsThisPeriod } = await supabase
    .from('leads')
    .select('*', { count: 'exact', head: true })
    .eq('workspace_id', workspaceId)
    .gte('created_at', periodStart.toISOString())

  // Período anterior para comparação
  const prevPeriodStart = new Date(periodStart)
  prevPeriodStart.setDate(prevPeriodStart.getDate() - days)

  const { count: leadsPrevPeriod } = await supabase
    .from('leads')
    .select('*', { count: 'exact', head: true })
    .eq('workspace_id', workspaceId)
    .gte('created_at', prevPeriodStart.toISOString())
    .lt('created_at', periodStart.toISOString())

  // Calcular variação
  const leadsChangePercent = leadsPrevPeriod
    ? ((leadsThisPeriod - leadsPrevPeriod) / leadsPrevPeriod) * 100
    : 0

  // Mensagens
  const { count: messagesSent } = await supabase
    .from('messages')
    .select('*', { count: 'exact', head: true })
    .eq('direction', 'outbound')
    .gte('sent_at', periodStart.toISOString())

  const { count: messagesReceived } = await supabase
    .from('messages')
    .select('*', { count: 'exact', head: true })
    .eq('direction', 'inbound')
    .gte('sent_at', periodStart.toISOString())

  // Leads por estágio
  const { data: leadsByStage } = await supabase
    .from('leads')
    .select('stage:kanban_stages(name), count:id.count()')
    .eq('workspace_id', workspaceId)
    .group('stage_id')

  return {
    period,
    total_leads: totalLeads || 0,
    leads_this_period: leadsThisPeriod || 0,
    leads_change_percent: leadsChangePercent,
    conversion_rate: await calculateConversionRate(workspaceId),
    conversion_change_percent: 0, // TODO: comparar com período anterior
    messages_sent: messagesSent || 0,
    messages_received: messagesReceived || 0,
    avg_time_to_conversion: await calculateAvgTimeToConversion(workspaceId),
    leads_by_stage: leadsByStage || []
  }
}

## Cache

```typescript
import NodeCache from 'node-cache'

const cache = new NodeCache({ stdTTL: 300 }) // 5 minutos

export async function getOverviewMetricsCached(
  workspaceId: string,
  period: string
) {
  const cacheKey = `overview:${workspaceId}:${period}`

  const cached = cache.get(cacheKey)
  if (cached) return cached

  const data = await getOverviewMetrics(workspaceId, period as any)
  cache.set(cacheKey, data)

  return data
}
```
