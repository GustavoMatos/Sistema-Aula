## Componentes

**MessageHistory.tsx:**
```typescript
export function MessageHistory({ leadId }: { leadId: string }) {
  const { messages, isLoading } = useMessages(leadId)
  const scrollRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    scrollRef.current?.scrollIntoView({ behavior: 'smooth' })
  }, [messages])

  return (
    <Card>
      <CardHeader>
        <CardTitle>Mensagens</CardTitle>
      </CardHeader>
      <CardContent className="max-h-96 overflow-y-auto">
        <div className="space-y-4">
          {messages.map((msg) => (
            <MessageBubble key={msg.id} message={msg} />
          ))}
          <div ref={scrollRef} />
        </div>
      </CardContent>
      <CardFooter>
        <MessageComposer leadId={leadId} />
      </CardFooter>
    </Card>
  )
}
```

**MessageBubble.tsx:**
```typescript
export function MessageBubble({ message }: { message: Message }) {
  const isOutbound = message.direction === 'outbound'

  return (
    <div className={cn(
      'flex',
      isOutbound ? 'justify-end' : 'justify-start'
    )}>
      <div className={cn(
        'max-w-[70%] rounded-lg p-3',
        isOutbound ? 'bg-blue-500 text-white' : 'bg-gray-100'
      )}>
        <p>{message.content}</p>
        <div className="flex items-center gap-1 text-xs mt-1 opacity-70">
          <span>{formatTime(message.sent_at)}</span>
          {message.is_automated && <Zap className="w-3 h-3" />}
        </div>
      </div>
    </div>
  )
}
```

## Hook useMessages

```typescript
export function useMessages(leadId: string) {
  const queryClient = useQueryClient()

  const query = useQuery({
    queryKey: ['messages', leadId],
    queryFn: () => api.get(`/leads/${leadId}/messages`)
  })

  // Realtime subscription
  useEffect(() => {
    const channel = supabase
      .channel(`messages:${leadId}`)
      .on('postgres_changes', {
        event: 'INSERT',
        schema: 'public',
        table: 'messages',
        filter: `lead_id=eq.${leadId}`
      }, (payload) => {
        queryClient.setQueryData(['messages', leadId], (old: Message[]) =>
          [...old, payload.new]
        )
      })
      .subscribe()

    return () => { channel.unsubscribe() }
  }, [leadId])

  return query
}
```
