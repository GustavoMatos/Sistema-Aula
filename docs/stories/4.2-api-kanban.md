## Notas Técnicas

**KanbanService:**
```typescript
export class KanbanService {
  async getData(workspaceId: string, filters: KanbanFilters) {
    // Get all stages
    const { data: stages } = await supabase
      .from('kanban_stages')
      .select('*')
      .eq('workspace_id', workspaceId)
      .order('position', { ascending: true })

    if (!stages?.length) return { stages: [] }

    // Build leads query
    let leadsQuery = supabase
      .from('leads')
      .select('*', { count: 'exact' })
      .eq('workspace_id', workspaceId)

    if (filters.tags?.length) {
      leadsQuery = leadsQuery.overlaps('tags', filters.tags)
    }

    if (filters.source) {
      leadsQuery = leadsQuery.eq('source', filters.source)
    }

    const { data: allLeads } = await leadsQuery

    // Group leads by stage
    const leadsPerStage = filters.leads_per_stage || 50

    const stagesWithLeads = stages.map(stage => {
      const stageLeads = (allLeads || [])
        .filter(lead => lead.stage_id === stage.id)
        .sort((a, b) => new Date(b.created_at).getTime() - new Date(a.created_at).getTime())

      return {
        ...stage,
        total_leads: stageLeads.length,
        leads: stageLeads.slice(0, leadsPerStage).map(lead => ({
          ...lead,
          days_in_stage: this.calculateDaysInStage(lead),
        })),
      }
    })

    return { stages: stagesWithLeads }
  }

  private calculateDaysInStage(lead: Lead): number {
    // Idealmente, buscar do histórico a última mudança de estágio
    // Simplificado: usar created_at
    const created = new Date(lead.created_at)
    const now = new Date()
    return Math.floor((now.getTime() - created.getTime()) / (1000 * 60 * 60 * 24))
  }
}
```

**Otimização - Query com contagem:**
```sql
-- Versão otimizada com subqueries
SELECT
  ks.*,
  (SELECT COUNT(*) FROM leads WHERE stage_id = ks.id) as total_leads,
  (
    SELECT json_agg(l.*)
    FROM (
      SELECT * FROM leads
      WHERE stage_id = ks.id
      ORDER BY created_at DESC
      LIMIT 50
    ) l
  ) as leads
FROM kanban_stages ks
WHERE ks.workspace_id = $1
ORDER BY ks.position;
```
