## Implementação

```typescript
// evolution.service.ts
export async function configureWebhook(instanceName: string): Promise<void> {
  const webhookUrl = `${process.env.BACKEND_URL}/api/webhooks/evolution`
  const secret = generateWebhookSecret()

  const response = await fetch(
    `${EVOLUTION_API_URL}/webhook/set/${instanceName}`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'apikey': EVOLUTION_API_KEY
      },
      body: JSON.stringify({
        url: webhookUrl,
        enabled: true,
        webhookByEvents: true,
        webhookBase64: false,
        events: [
          'MESSAGES_UPSERT',
          'CONNECTION_UPDATE',
          'QRCODE_UPDATED'
        ]
      })
    }
  )

  if (!response.ok) {
    throw new Error('Failed to configure webhook')
  }

  // Salvar secret no banco
  await updateInstanceWebhookSecret(instanceName, secret)
}

function generateWebhookSecret(): string {
  return crypto.randomBytes(32).toString('hex')
}
```
