## Notas Técnicas

**LeadDetail.tsx:**
```typescript
export default function LeadDetail() {
  const { id } = useParams<{ id: string }>()
  const navigate = useNavigate()

  const { data: lead, isLoading } = useLead(id!)
  const { data: history } = useLeadHistory(id!)

  if (isLoading) return <LoadingSpinner />
  if (!lead) return <NotFound />

  return (
    <div className="container py-6 space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-4">
          <Button variant="ghost" onClick={() => navigate(-1)}>
            <ArrowLeft className="w-4 h-4 mr-2" />
            Voltar
          </Button>
          <h1 className="text-2xl font-bold">{lead.name}</h1>
        </div>
        <LeadActionsBar lead={lead} />
      </div>

      {/* Main content */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
        <div className="md:col-span-2 space-y-6">
          <LeadInfoCard lead={lead} />
          <LeadTagsSection lead={lead} />
          <LeadNotesSection lead={lead} />
          <LeadHistoryTimeline history={history} />
        </div>

        <div className="space-y-6">
          <LeadQuickActions lead={lead} />
          <AutomationToggle lead={lead} />
        </div>
      </div>
    </div>
  )
}
```

**LeadHistoryTimeline.tsx:**
```typescript
const actionIcons = {
  lead_created: Plus,
  stage_change: ArrowRight,
  message_sent: MessageSquare,
  message_received: MessageSquare,
  note_added: FileText,
}

const actionLabels = {
  lead_created: 'Lead criado',
  stage_change: 'Movido para',
  message_sent: 'Mensagem enviada',
  message_received: 'Mensagem recebida',
  note_added: 'Nota adicionada',
}

export function LeadHistoryTimeline({ history }: Props) {
  return (
    <Card>
      <CardHeader>
        <CardTitle>Histórico</CardTitle>
      </CardHeader>
      <CardContent>
        <div className="relative">
          {history.map((item, index) => (
            <div key={item.id} className="flex gap-4 pb-6">
              <div className="flex flex-col items-center">
                <div className="w-8 h-8 rounded-full bg-muted flex items-center justify-center">
                  {React.createElement(actionIcons[item.action] || Circle, {
                    className: 'w-4 h-4',
                  })}
                </div>
                {index < history.length - 1 && (
                  <div className="w-px h-full bg-border" />
                )}
              </div>
              <div className="flex-1">
                <p className="text-sm font-medium">
                  {actionLabels[item.action]}
                  {item.to_stage && ` ${item.to_stage.name}`}
                </p>
                {item.performed_by && (
                  <p className="text-xs text-muted-foreground">
                    por {item.performed_by.full_name}
                  </p>
                )}
                <p className="text-xs text-muted-foreground">
                  {formatDistanceToNow(new Date(item.created_at), {
                    addSuffix: true,
                    locale: ptBR,
                  })}
                </p>
              </div>
            </div>
          ))}
        </div>
      </CardContent>
    </Card>
  )
}
```
