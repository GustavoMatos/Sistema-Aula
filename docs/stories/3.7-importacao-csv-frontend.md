**ImportCSVModal.tsx:**
```typescript
export function ImportCSVModal({ open, onClose }: Props) {
  const [file, setFile] = useState<File | null>(null)
  const [preview, setPreview] = useState<string[][] | null>(null)
  const [result, setResult] = useState<ImportResult | null>(null)

  const importMutation = useMutation({
    mutationFn: (file: File) => api.leads.importCSV(file),
    onSuccess: (data) => setResult(data),
  })

  const handleFileSelect = async (file: File) => {
    setFile(file)

    // Parse preview
    const text = await file.text()
    const lines = text.split('\n').slice(0, 6) // Header + 5 rows
    const rows = lines.map(line => line.split(',').map(cell => cell.trim()))
    setPreview(rows)
  }

  const handleImport = () => {
    if (file) {
      importMutation.mutate(file)
    }
  }

  const downloadTemplate = () => {
    const template = 'name,phone,email,company,source\nJoão Silva,5511999999999,joao@email.com,Empresa X,formulário'
    const blob = new Blob([template], { type: 'text/csv' })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = 'template-leads.csv'
    a.click()
  }

  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent className="max-w-2xl">
        {result ? (
          <ImportResults result={result} onClose={onClose} />
        ) : (
          <>
            <DialogHeader>
              <DialogTitle>Importar Leads</DialogTitle>
            </DialogHeader>

            {!file ? (
              <FileDropzone onFileSelect={handleFileSelect} />
            ) : (
              <CSVPreview
                fileName={file.name}
                preview={preview}
                onRemove={() => setFile(null)}
              />
            )}

            <div className="flex items-center justify-between pt-4">
              <Button variant="link" onClick={downloadTemplate}>
                📥 Baixar template CSV
              </Button>

              {file && (
                <div className="flex gap-2">
                  <Button variant="outline" onClick={onClose}>
                    Cancelar
                  </Button>
                  <Button
                    onClick={handleImport}
                    disabled={importMutation.isPending}
                  >
                    {importMutation.isPending ? (
                      <>
                        <Loader2 className="w-4 h-4 mr-2 animate-spin" />
                        Importando...
                      </>
                    ) : (
                      'Importar'
                    )}
                  </Button>
                </div>
              )}
            </div>
          </>
        )}
      </DialogContent>
    </Dialog>
  )
}
```

**FileDropzone.tsx:**
```typescript
export function FileDropzone({ onFileSelect }: Props) {
  const [isDragging, setIsDragging] = useState(false)

  const handleDrop = (e: React.DragEvent) => {
    e.preventDefault()
    setIsDragging(false)

    const file = e.dataTransfer.files[0]
    if (file?.name.endsWith('.csv')) {
      onFileSelect(file)
    } else {
      toast.error('Apenas arquivos CSV são permitidos')
    }
  }

  return (
    <div
      className={cn(
        'border-2 border-dashed rounded-lg p-12 text-center',
        'transition-colors cursor-pointer',
        isDragging ? 'border-primary bg-primary/5' : 'border-muted-foreground/25'
      )}
      onDragOver={(e) => { e.preventDefault(); setIsDragging(true) }}
      onDragLeave={() => setIsDragging(false)}
      onDrop={handleDrop}
      onClick={() => document.getElementById('csv-input')?.click()}
    >
      <input
        id="csv-input"
        type="file"
        accept=".csv"
        className="hidden"
        onChange={(e) => e.target.files?.[0] && onFileSelect(e.target.files[0])}
      />
      <FileSpreadsheet className="w-12 h-12 mx-auto mb-4 text-muted-foreground" />
      <p className="text-lg">Arraste seu arquivo CSV aqui</p>
      <p className="text-sm text-muted-foreground">ou clique para selecionar</p>
    </div>
  )
}
```
