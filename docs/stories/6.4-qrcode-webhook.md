## Payload QRCODE_UPDATED

```json
{
  "event": "qrcode.updated",
  "instance": "minha-instancia",
  "data": {
    "qrcode": "base64_ou_pairing_code"
  }
}
```

---

## Implementação

```typescript
// qrcode-handler.ts
export async function handleQRCodeUpdated(instance: string, data: any) {
  const { qrcode } = data

  // Buscar instância pelo nome
  const whatsappInstance = await findInstanceByName(instance)

  if (!whatsappInstance) {
    console.warn(`[QRCode] Instance not found: ${instance}`)
    return
  }

  // Atualizar QR code
  await updateInstanceQRCode(whatsappInstance.id, qrcode)

  // Atualizar status para 'awaiting_scan'
  await updateInstanceStatus(whatsappInstance.id, 'awaiting_scan')

  console.log(`[QRCode] Updated for ${instance}`)
}
```

## Realtime Event

```typescript
// O frontend se inscreve em:
supabase
  .channel('whatsapp-instances')
  .on('postgres_changes', {
    event: 'UPDATE',
    schema: 'public',
    table: 'whatsapp_instances',
    filter: `id=eq.${instanceId}`
  }, (payload) => {
    // Atualizar QR code na UI
    setQRCode(payload.new.qr_code)
  })
  .subscribe()
```
