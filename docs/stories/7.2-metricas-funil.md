## Implementação

```typescript
// analytics.service.ts
export async function getFunnelMetrics(
  workspaceId: string,
  period: '7d' | '30d' | '90d'
): Promise<FunnelMetrics> {
  // Buscar estágios ordenados
  const { data: stages } = await supabase
    .from('kanban_stages')
    .select('id, name, position, is_final')
    .eq('workspace_id', workspaceId)
    .order('position')

  if (!stages) return { stages: [], total_conversion: 0 }

  const funnelStages = []
  let previousCount = 0

  for (let i = 0; i < stages.length; i++) {
    const stage = stages[i]
    const nextStage = stages[i + 1]

    // Contar leads neste estágio
    const { count: currentCount } = await supabase
      .from('leads')
      .select('*', { count: 'exact', head: true })
      .eq('stage_id', stage.id)

    // Calcular conversão para próximo
    let conversionToNext = 0
    if (nextStage && currentCount) {
      const { count: movedCount } = await supabase
        .from('lead_history')
        .select('*', { count: 'exact', head: true })
        .eq('from_stage_id', stage.id)
        .eq('to_stage_id', nextStage.id)

      conversionToNext = (movedCount / currentCount) * 100
    }

    // Tempo médio no estágio
    const avgTimeHours = await calculateAvgTimeInStage(stage.id)

    // Leads perdidos (movidos para estágio "Perdido")
    const dropped = await countDroppedFromStage(stage.id, workspaceId)

    funnelStages.push({
      name: stage.name,
      count: currentCount || 0,
      conversion_to_next: conversionToNext,
      avg_time_hours: avgTimeHours,
      dropped
    })

    previousCount = currentCount || 0
  }

  // Conversão total
  const firstStageCount = funnelStages[0]?.count || 0
  const wonCount = await countWonLeads(workspaceId)
  const totalConversion = firstStageCount
    ? (wonCount / firstStageCount) * 100
    : 0

  return {
    stages: funnelStages,
    total_conversion: totalConversion
  }
}
```

## Queries Auxiliares

```typescript
async function calculateAvgTimeInStage(stageId: string): Promise<number> {
  const { data } = await supabase.rpc('avg_time_in_stage', {
    stage_id: stageId
  })
  return data || 0
}

async function countDroppedFromStage(
  stageId: string,
  workspaceId: string
): Promise<number> {
  // Buscar estágio "Perdido"
  const { data: lostStage } = await supabase
    .from('kanban_stages')
    .select('id')
    .eq('workspace_id', workspaceId)
    .ilike('name', '%perdido%')
    .single()

  if (!lostStage) return 0

  const { count } = await supabase
    .from('lead_history')
    .select('*', { count: 'exact', head: true })
    .eq('from_stage_id', stageId)
    .eq('to_stage_id', lostStage.id)

  return count || 0
}

async function countWonLeads(workspaceId: string): Promise<number> {
  const { count } = await supabase
    .from('leads')
    .select('*, stage:kanban_stages!inner(*)', { count: 'exact', head: true })
    .eq('workspace_id', workspaceId)
    .eq('stage.is_final', true)
    .ilike('stage.name', '%ganho%')

  return count || 0
}
```

## SQL Function para Tempo Médio

```sql
CREATE OR REPLACE FUNCTION avg_time_in_stage(stage_id UUID)
RETURNS NUMERIC AS $$
DECLARE
  avg_hours NUMERIC;
BEGIN
  SELECT AVG(
    EXTRACT(EPOCH FROM (
      COALESCE(
        (SELECT MIN(created_at) FROM lead_history
         WHERE lead_id = l.id AND from_stage_id = stage_id),
        NOW()
      ) - l.created_at
    )) / 3600
  )
  INTO avg_hours
  FROM leads l
  WHERE l.stage_id = stage_id;

  RETURN COALESCE(avg_hours, 0);
END;
$$ LANGUAGE plpgsql;
```
