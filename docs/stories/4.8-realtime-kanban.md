## Notas Técnicas

**useLeadsRealtime.ts:**
```typescript
import { useEffect, useState } from 'react'
import { useQueryClient } from '@tanstack/react-query'
import { supabase } from '@/lib/supabase'
import { RealtimeChannel } from '@supabase/supabase-js'

export function useLeadsRealtime() {
  const queryClient = useQueryClient()
  const [isConnected, setIsConnected] = useState(true)

  useEffect(() => {
    let channel: RealtimeChannel

    const setupRealtime = () => {
      channel = supabase
        .channel('leads-changes')
        .on(
          'postgres_changes',
          {
            event: '*',
            schema: 'public',
            table: 'leads',
          },
          (payload) => {
            handleChange(payload)
          }
        )
        .subscribe((status) => {
          setIsConnected(status === 'SUBSCRIBED')
        })
    }

    const handleChange = (payload: any) => {
      const { eventType, new: newRecord, old: oldRecord } = payload

      switch (eventType) {
        case 'INSERT':
          // Invalidate to refetch with new lead
          queryClient.invalidateQueries({ queryKey: ['kanban'] })
          toast.info(`Novo lead: ${newRecord.name}`)
          break

        case 'UPDATE':
          // Update cache directly for smooth UX
          queryClient.setQueryData(['kanban'], (old: any) => {
            if (!old) return old
            return updateLeadInKanban(old, newRecord)
          })
          break

        case 'DELETE':
          // Remove from cache
          queryClient.setQueryData(['kanban'], (old: any) => {
            if (!old) return old
            return removeLeadFromKanban(old, oldRecord.id)
          })
          break
      }
    }

    setupRealtime()

    return () => {
      if (channel) {
        supabase.removeChannel(channel)
      }
    }
  }, [queryClient])

  return { isConnected }
}

// Helper functions
function updateLeadInKanban(data: KanbanData, updatedLead: Lead): KanbanData {
  return {
    stages: data.stages.map(stage => {
      // Check if lead moved stages
      const wasInThisStage = stage.leads.some(l => l.id === updatedLead.id)
      const shouldBeInThisStage = stage.id === updatedLead.stage_id

      if (wasInThisStage && !shouldBeInThisStage) {
        // Remove from this stage
        return {
          ...stage,
          leads: stage.leads.filter(l => l.id !== updatedLead.id),
          total_leads: stage.total_leads - 1,
        }
      }

      if (!wasInThisStage && shouldBeInThisStage) {
        // Add to this stage
        return {
          ...stage,
          leads: [updatedLead, ...stage.leads],
          total_leads: stage.total_leads + 1,
        }
      }

      if (wasInThisStage && shouldBeInThisStage) {
        // Update in place
        return {
          ...stage,
          leads: stage.leads.map(l =>
            l.id === updatedLead.id ? updatedLead : l
          ),
        }
      }

      return stage
    }),
  }
}

function removeLeadFromKanban(data: KanbanData, leadId: string): KanbanData {
  return {
    stages: data.stages.map(stage => ({
      ...stage,
      leads: stage.leads.filter(l => l.id !== leadId),
      total_leads: stage.leads.some(l => l.id === leadId)
        ? stage.total_leads - 1
        : stage.total_leads,
    })),
  }
}
```

function ConnectionIndicator({ isConnected }: { isConnected: boolean }) {
  if (isConnected) return null

  return (
    <div className="fixed bottom-4 right-4 bg-yellow-100 border border-yellow-300 rounded-lg p-3 flex items-center gap-2 shadow-lg">
      <Loader2 className="w-4 h-4 animate-spin text-yellow-600" />
      <span className="text-sm text-yellow-800">Reconectando...</span>
    </div>
  )
}

## Eventos do Supabase Realtime

| Evento | Descrição | Ação |
|--------|-----------|------|
| `INSERT` | Novo lead criado | Invalidar query + toast |
| `UPDATE` | Lead atualizado | Atualizar cache diretamente |
| `DELETE` | Lead removido | Remover do cache |
