## Componente FunnelChart

```typescript
import { ResponsiveContainer, FunnelChart as ReChart, Funnel, Tooltip, Cell, LabelList } from 'recharts'

interface FunnelChartProps {
  data: FunnelData | undefined
  loading?: boolean
  onStageClick?: (stageName: string) => void
}

export function FunnelChart({ data, loading, onStageClick }: FunnelChartProps) {
  if (loading || !data) {
    return (
      <Card>
        <CardHeader>
          <CardTitle>Funil de Conversão</CardTitle>
        </CardHeader>
        <CardContent>
          <Skeleton className="h-64 w-full" />
        </CardContent>
      </Card>
    )
  }

  const chartData = data.stages.map((stage, index) => ({
    name: stage.name,
    value: stage.count,
    percent: index === 0
      ? 100
      : (stage.count / data.stages[0].count * 100).toFixed(1),
    conversion: stage.conversion_to_next,
    avgTime: stage.avg_time_hours,
    dropped: stage.dropped,
    fill: getStageColor(index)
  }))

  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center justify-between">
          Funil de Conversão
          <span className="text-sm font-normal text-muted-foreground">
            Taxa total: {data.total_conversion.toFixed(1)}%
          </span>
        </CardTitle>
      </CardHeader>
      <CardContent>
        <ResponsiveContainer width="100%" height={300}>
          <ReChart>
            <Tooltip content={<CustomTooltip />} />
            <Funnel
              data={chartData}
              dataKey="value"
              isAnimationActive
              animationDuration={800}
              onClick={(data) => onStageClick?.(data.name)}
            >
              {chartData.map((entry, index) => (
                <Cell
                  key={`cell-${index}`}
                  fill={entry.fill}
                  className="cursor-pointer hover:opacity-80 transition-opacity"
                />
              ))}
              <LabelList
                position="center"
                content={renderLabel}
              />
            </Funnel>
          </ReChart>
        </ResponsiveContainer>
      </CardContent>
    </Card>
  )
}

function CustomTooltip({ active, payload }: any) {
  if (!active || !payload?.[0]) return null

  const data = payload[0].payload

  return (
    <div className="bg-white p-3 rounded-lg shadow-lg border">
      <p className="font-medium">{data.name}</p>
      <div className="text-sm text-muted-foreground mt-1 space-y-1">
        <p>Leads: {data.value} ({data.percent}%)</p>
        <p>Conversão: {data.conversion.toFixed(1)}%</p>
        <p>Tempo médio: {formatHours(data.avgTime)}</p>
        <p>Perdidos: {data.dropped}</p>
      </div>
    </div>
  )
}

function renderLabel({ x, y, width, value, name }: any) {
  return (
    <text
      x={x + width / 2}
      y={y + 20}
      fill="#fff"
      textAnchor="middle"
      dominantBaseline="central"
      className="text-sm font-medium"
    >
      {name}: {value}
    </text>
  )
}

function getStageColor(index: number): string {
  const colors = [
    '#3b82f6', // blue
    '#8b5cf6', // violet
    '#ec4899', // pink
    '#f97316', // orange
    '#22c55e', // green
    '#6b7280', // gray
  ]
  return colors[index % colors.length]
}

function formatHours(hours: number): string {
  if (hours < 24) return `${hours.toFixed(0)}h`
  const days = Math.floor(hours / 24)
  return `${days}d`
}
```

## Alternativa: Funil Horizontal (CSS)

```typescript
export function SimpleFunnel({ data }: { data: FunnelData }) {
  const maxCount = data.stages[0]?.count || 1

  return (
    <div className="space-y-2">
      {data.stages.map((stage, index) => {
        const width = (stage.count / maxCount) * 100
        const percent = index === 0
          ? 100
          : (stage.count / maxCount * 100)

        return (
          <div key={stage.name} className="relative">
            <div
              className="h-10 rounded transition-all duration-500"
              style={{
                width: `${width}%`,
                backgroundColor: getStageColor(index)
              }}
            />
            <div className="absolute inset-0 flex items-center px-3">
              <span className="text-white font-medium">
                {stage.name}: {stage.count} ({percent.toFixed(0)}%)
              </span>
            </div>
          </div>
        )
      })}
    </div>
  )
}
```
