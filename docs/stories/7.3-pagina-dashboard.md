export function DashboardPage() {
  const [period, setPeriod] = useState<'7d' | '30d' | '90d'>('30d')
  const { overview, funnel, activity, isLoading } = useDashboard(period)

  return (
    <div className="container mx-auto p-6 space-y-6">
      {/* Header */}
      <div className="flex justify-between items-center">
        <h1 className="text-2xl font-bold">Dashboard</h1>
        <PeriodSelector value={period} onChange={setPeriod} />
      </div>

      {/* KPI Cards */}
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-4">
        <MetricCard
          title="Total de Leads"
          value={overview?.total_leads}
          change={overview?.leads_change_percent}
          loading={isLoading}
          icon={<Users />}
        />
        <MetricCard
          title="Novos Leads"
          value={overview?.leads_this_period}
          change={20}
          loading={isLoading}
          icon={<UserPlus />}
        />
        <MetricCard
          title="Taxa de Conversão"
          value={`${overview?.conversion_rate}%`}
          change={overview?.conversion_change_percent}
          loading={isLoading}
          icon={<TrendingUp />}
        />
        <MetricCard
          title="Mensagens Enviadas"
          value={overview?.messages_sent}
          change={10}
          loading={isLoading}
          icon={<MessageSquare />}
        />
      </div>

      {/* Charts Row */}
      <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
        <div className="lg:col-span-2">
          <FunnelChart data={funnel} loading={isLoading} />
        </div>
        <WhatsAppStatusCard />
      </div>

      {/* Lists Row */}
      <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
        <RecentLeadsList />
        <ActivityFeed activities={activity} />
      </div>
    </div>
  )
}

## Hook useDashboard

```typescript
export function useDashboard(period: string) {
  const overview = useQuery({
    queryKey: ['analytics', 'overview', period],
    queryFn: () => api.get(`/analytics/overview?period=${period}`),
    refetchInterval: 5 * 60 * 1000 // 5 minutos
  })

  const funnel = useQuery({
    queryKey: ['analytics', 'funnel', period],
    queryFn: () => api.get(`/analytics/funnel?period=${period}`),
    refetchInterval: 5 * 60 * 1000
  })

  const activity = useQuery({
    queryKey: ['analytics', 'activity'],
    queryFn: () => api.get('/analytics/activity'),
    refetchInterval: 60 * 1000 // 1 minuto
  })

  return {
    overview: overview.data,
    funnel: funnel.data,
    activity: activity.data,
    isLoading: overview.isLoading || funnel.isLoading
  }
}
```
