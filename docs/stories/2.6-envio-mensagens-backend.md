## Endpoint

### POST /api/messages/send

**Request:**
```json
{
  "lead_id": "uuid",
  "type": "text",
  "content": "Olá, tudo bem?",
  "media_url": null,
  "caption": null
}
```

**Request (imagem):**
```json
{
  "lead_id": "uuid",
  "type": "image",
  "content": null,
  "media_url": "https://example.com/image.png",
  "caption": "Confira nossa oferta!"
}
```

**Response 201:**
```json
{
  "id": "uuid",
  "lead_id": "uuid",
  "direction": "outbound",
  "content_type": "text",
  "content": "Olá, tudo bem?",
  "status": "sent",
  "whatsapp_message_id": "BAE594145F4C59B4",
  "sent_at": "2024-03-28T10:05:00Z"
}
```

## Notas Técnicas

**MessagesService:**
```typescript
export class MessagesService {
  private evolution: EvolutionService

  async send(data: SendMessageDTO): Promise<Message> {
    // Buscar lead e instância
    const { data: lead } = await supabase
      .from('leads')
      .select('*, whatsapp_instances(*)')
      .eq('id', data.lead_id)
      .single()

    if (!lead) throw new AppError('Lead not found', 404)

    const instance = lead.whatsapp_instances?.[0]
    if (!instance) throw new AppError('No WhatsApp instance', 400)

    // Enviar via Evolution
    let response: MessageResponse
    if (data.type === 'text') {
      response = await this.evolution.sendText(instance.instance_name, {
        number: lead.phone,
        text: data.content,
      })
    } else if (data.type === 'image') {
      response = await this.evolution.sendMedia(instance.instance_name, {
        number: lead.phone,
        mediaType: 'image',
        mimeType: 'image/png',
        mediaUrl: data.media_url!,
        caption: data.caption,
      })
    }

    // Salvar mensagem
    const { data: message } = await supabase
      .from('messages')
      .insert({
        lead_id: data.lead_id,
        instance_id: instance.id,
        direction: 'outbound',
        content_type: data.type,
        content: data.content || data.caption,
        media_url: data.media_url,
        status: 'sent',
        whatsapp_message_id: response.key.id,
        sent_at: new Date().toISOString(),
      })
      .select()
      .single()

    // Atualizar lead
    await supabase
      .from('leads')
      .update({ last_contact_at: new Date().toISOString() })
      .eq('id', data.lead_id)

    return message
  }
}
```

function validatePhoneNumber(phone: string): boolean {
  // Formato: código do país + número (sem +)
  // Ex: 5511999999999
  return /^\d{10,15}$/.test(phone)
}
