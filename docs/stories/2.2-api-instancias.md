## Notas Técnicas

**Controller Pattern:**
```typescript
export class WhatsAppController {
  private instanceService: WhatsAppInstanceService
  private evolutionService: EvolutionService

  create = async (req: Request, res: Response, next: NextFunction) => {
    try {
      const workspaceId = req.user!.workspace_id
      const { name } = req.body

      // Criar na Evolution API
      const evolutionResponse = await this.evolutionService.createInstance(name)

      // Salvar no banco
      const instance = await this.instanceService.create({
        workspace_id: workspaceId,
        instance_name: name,
        api_key: evolutionResponse.hash.apikey,
      })

      // Configurar webhook
      await this.evolutionService.setWebhook(name, {
        url: `${config.webhookBaseUrl}/api/webhooks/evolution`,
        events: ['MESSAGES_UPSERT', 'CONNECTION_UPDATE', 'QRCODE_UPDATED']
      })

      res.status(201).json(instance)
    } catch (error) {
      next(error)
    }
  }
}
```
