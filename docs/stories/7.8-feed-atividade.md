## Tipos de Atividade

```typescript
type ActivityType =
  | 'message_received'
  | 'message_sent'
  | 'stage_changed'
  | 'lead_created'
  | 'followup_sent'
  | 'lead_won'
  | 'lead_lost'

interface Activity {
  id: string
  type: ActivityType
  lead_id: string
  lead_name: string
  details: Record<string, any>
  created_at: string
}
```

```typescript
export function ActivityFeed() {
  const { activities, isLoading } = useActivityFeed()

  if (isLoading) {
    return (
      <Card>
        <CardHeader>
          <CardTitle>Atividade Recente</CardTitle>
        </CardHeader>
        <CardContent>
          {[1, 2, 3, 4, 5].map((i) => (
            <Skeleton key={i} className="h-10 mb-2" />
          ))}
        </CardContent>
      </Card>
    )
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center gap-2">
          <Activity className="w-5 h-5" />
          Atividade Recente
        </CardTitle>
      </CardHeader>
      <CardContent>
        <div className="space-y-1">
          {activities.length === 0 ? (
            <p className="text-muted-foreground text-sm py-4">
              Nenhuma atividade recente
            </p>
          ) : (
            activities.map((activity) => (
              <ActivityItem key={activity.id} activity={activity} />
            ))
          )}
        </div>
      </CardContent>
      <CardFooter>
        <Link
          to="/activity"
          className="text-sm text-blue-600 hover:underline"
        >
          Ver toda atividade →
        </Link>
      </CardFooter>
    </Card>
  )
}
```

## Componente ActivityItem

```typescript
export function ActivityItem({ activity }: { activity: Activity }) {
  const { icon, color, text } = getActivityConfig(activity)

  return (
    <div className="flex items-start gap-3 p-2 rounded-lg hover:bg-muted/30">
      {/* Icon */}
      <div className={cn(
        'w-8 h-8 rounded-full flex items-center justify-center',
        color
      )}>
        {icon}
      </div>

      {/* Content */}
      <div className="flex-1 min-w-0">
        <p className="text-sm">
          <span className="font-medium">{activity.lead_name}</span>
          {' '}{text}
        </p>
        <p className="text-xs text-muted-foreground">
          {formatRelativeTime(activity.created_at)}
        </p>
      </div>
    </div>
  )
}

function getActivityConfig(activity: Activity) {
  const configs: Record<ActivityType, { icon: ReactNode; color: string; text: string }> = {
    message_received: {
      icon: <MessageCircle className="w-4 h-4 text-white" />,
      color: 'bg-green-500',
      text: 'respondeu'
    },
    message_sent: {
      icon: <Send className="w-4 h-4 text-white" />,
      color: 'bg-blue-500',
      text: 'recebeu mensagem'
    },
    stage_changed: {
      icon: <ArrowRight className="w-4 h-4 text-white" />,
      color: 'bg-purple-500',
      text: `moveu para ${activity.details.to_stage}`
    },
    lead_created: {
      icon: <UserPlus className="w-4 h-4 text-white" />,
      color: 'bg-cyan-500',
      text: 'foi criado'
    },
    followup_sent: {
      icon: <Zap className="w-4 h-4 text-white" />,
      color: 'bg-yellow-500',
      text: 'recebeu follow-up automático'
    },
    lead_won: {
      icon: <Trophy className="w-4 h-4 text-white" />,
      color: 'bg-emerald-500',
      text: 'foi fechado (ganho)'
    },
    lead_lost: {
      icon: <XCircle className="w-4 h-4 text-white" />,
      color: 'bg-red-500',
      text: 'foi perdido'
    }
  }

  return configs[activity.type] || {
    icon: <Circle className="w-4 h-4 text-white" />,
    color: 'bg-gray-500',
    text: 'teve uma atividade'
  }
}
```

## Hook useActivityFeed

```typescript
export function useActivityFeed(limit = 5) {
  const queryClient = useQueryClient()

  const query = useQuery({
    queryKey: ['activity-feed', limit],
    queryFn: async () => {
      const { data, error } = await supabase
        .from('lead_history')
        .select(`
          id,
          action,
          changes,
          created_at,
          lead:leads(id, name)
        `)
        .order('created_at', { ascending: false })
        .limit(limit)

      if (error) throw error

      // Transformar para formato de Activity
      return data.map(item => ({
        id: item.id,
        type: item.action as ActivityType,
        lead_id: item.lead.id,
        lead_name: item.lead.name,
        details: item.changes || {},
        created_at: item.created_at
      }))
    }
  })

  // Realtime subscription
  useEffect(() => {
    const channel = supabase
      .channel('activity-feed')
      .on('postgres_changes', {
        event: 'INSERT',
        schema: 'public',
        table: 'lead_history'
      }, () => {
        queryClient.invalidateQueries({ queryKey: ['activity-feed'] })
      })
      .subscribe()

    return () => { channel.unsubscribe() }
  }, [queryClient])

  return {
    activities: query.data || [],
    isLoading: query.isLoading
  }
}
```
