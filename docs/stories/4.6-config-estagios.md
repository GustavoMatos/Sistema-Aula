## Notas Técnicas

**StagesConfig.tsx:**
```typescript
import { DndContext, closestCenter, DragEndEvent } from '@dnd-kit/core'
import { SortableContext, verticalListSortingStrategy } from '@dnd-kit/sortable'

export default function StagesConfig() {
  const { data: stages, isLoading } = useStages()
  const reorderMutation = useReorderStages()
  const [localStages, setLocalStages] = useState<Stage[]>([])

  useEffect(() => {
    if (stages) setLocalStages(stages)
  }, [stages])

  const handleDragEnd = async (event: DragEndEvent) => {
    const { active, over } = event
    if (!over || active.id === over.id) return

    const oldIndex = localStages.findIndex(s => s.id === active.id)
    const newIndex = localStages.findIndex(s => s.id === over.id)

    const newOrder = arrayMove(localStages, oldIndex, newIndex)
    setLocalStages(newOrder)

    await reorderMutation.mutateAsync(newOrder.map(s => s.id))
  }

  return (
    <div className="container py-6 max-w-2xl">
      <h1 className="text-2xl font-bold mb-6">Estágios do Kanban</h1>

      <p className="text-muted-foreground mb-4">
        Arraste para reordenar os estágios
      </p>

      <DndContext collisionDetection={closestCenter} onDragEnd={handleDragEnd}>
        <SortableContext items={localStages} strategy={verticalListSortingStrategy}>
          <div className="space-y-2">
            {localStages.map(stage => (
              <SortableStageItem key={stage.id} stage={stage} />
            ))}
          </div>
        </SortableContext>
      </DndContext>

      <AddStageForm onSuccess={() => {}} />
    </div>
  )
}
```

**SortableStageItem.tsx:**
```typescript
import { useSortable } from '@dnd-kit/sortable'
import { CSS } from '@dnd-kit/utilities'

export function SortableStageItem({ stage }: Props) {
  const {
    attributes,
    listeners,
    setNodeRef,
    transform,
    transition,
  } = useSortable({ id: stage.id })

  const style = {
    transform: CSS.Transform.toString(transform),
    transition,
  }

  const updateStage = useUpdateStage()
  const deleteStage = useDeleteStage()

  return (
    <div
      ref={setNodeRef}
      style={style}
      className="flex items-center gap-3 p-3 bg-card border rounded-lg"
    >
      {/* Drag handle */}
      <button {...attributes} {...listeners} className="cursor-grab">
        <GripVertical className="w-5 h-5 text-muted-foreground" />
      </button>

      {/* Color picker */}
      <ColorPicker
        color={stage.color}
        onChange={(color) => updateStage.mutate({ id: stage.id, color })}
      />

      {/* Name input */}
      <Input
        value={stage.name}
        onChange={(e) => updateStage.mutate({ id: stage.id, name: e.target.value })}
        className="flex-1"
      />

      {/* Is final checkbox */}
      <label className="flex items-center gap-2 text-sm">
        <Checkbox
          checked={stage.is_final}
          onCheckedChange={(checked) =>
            updateStage.mutate({ id: stage.id, is_final: !!checked })
          }
        />
        Final
      </label>

      {/* Delete button */}
      <DeleteStageDialog
        stage={stage}
        onConfirm={() => deleteStage.mutate(stage.id)}
      />
    </div>
  )
}
```

**ColorPicker.tsx:**
```typescript
const colors = [
  '#6366f1', '#8b5cf6', '#ec4899', '#f59e0b',
  '#10b981', '#22c55e', '#ef4444', '#3b82f6',
]

export function ColorPicker({ color, onChange }: Props) {
  return (
    <Popover>
      <PopoverTrigger asChild>
        <button
          className="w-6 h-6 rounded-full border-2 border-white shadow"
          style={{ backgroundColor: color }}
        />
      </PopoverTrigger>
      <PopoverContent className="w-auto p-2">
        <div className="grid grid-cols-4 gap-2">
          {colors.map(c => (
            <button
              key={c}
              className={cn(
                'w-8 h-8 rounded-full',
                c === color && 'ring-2 ring-offset-2 ring-primary'
              )}
              style={{ backgroundColor: c }}
              onClick={() => onChange(c)}
            />
          ))}
        </div>
      </PopoverContent>
    </Popover>
  )
}
```
