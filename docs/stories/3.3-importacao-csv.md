## Endpoint

### POST /api/leads/import

**Request:**
- Content-Type: `multipart/form-data`
- Field: `file` (CSV)

**Formato CSV esperado:**
```csv
name,phone,email,company,source
João Silva,5511999999999,joao@email.com,Empresa X,formulário
Maria Santos,5521988888888,maria@email.com,Empresa Y,indicação
```

**Response 200:**
```json
{
  "imported": 45,
  "duplicates": 5,
  "errors": [
    { "row": 12, "field": "phone", "error": "Telefone inválido: abc123" },
    { "row": 25, "field": "name", "error": "Nome é obrigatório" }
  ],
  "total_rows": 52
}
```

## Notas Técnicas

**ImportService:**
```typescript
import { parse } from 'csv-parse/sync'

export class ImportService {
  async importCSV(
    workspaceId: string,
    userId: string,
    file: Express.Multer.File
  ): Promise<ImportResult> {
    const content = file.buffer.toString('utf-8')
    const records = parse(content, {
      columns: true,
      skip_empty_lines: true,
      trim: true,
    })

    if (records.length > 1000) {
      throw new AppError('Máximo de 1000 leads por importação', 400)
    }

    // Buscar primeiro estágio
    const { data: firstStage } = await supabase
      .from('kanban_stages')
      .select('id')
      .eq('workspace_id', workspaceId)
      .order('position', { ascending: true })
      .limit(1)
      .single()

    // Buscar telefones existentes
    const phones = records.map(r => r.phone).filter(Boolean)
    const { data: existing } = await supabase
      .from('leads')
      .select('phone')
      .eq('workspace_id', workspaceId)
      .in('phone', phones)

    const existingPhones = new Set(existing?.map(e => e.phone) || [])

    const result: ImportResult = {
      imported: 0,
      duplicates: 0,
      errors: [],
      total_rows: records.length,
    }

    const toInsert: any[] = []

    for (let i = 0; i < records.length; i++) {
      const row = records[i]
      const rowNum = i + 2 // +2 porque linha 1 é header

      // Validar nome
      if (!row.name?.trim()) {
        result.errors.push({ row: rowNum, field: 'name', error: 'Nome é obrigatório' })
        continue
      }

      // Validar telefone
      const phone = row.phone?.replace(/\D/g, '')
      if (!phone || phone.length < 10 || phone.length > 15) {
        result.errors.push({ row: rowNum, field: 'phone', error: `Telefone inválido: ${row.phone}` })
        continue
      }

      // Verificar duplicado
      if (existingPhones.has(phone)) {
        result.duplicates++
        continue
      }

      existingPhones.add(phone) // Evitar duplicados no mesmo arquivo

      toInsert.push({
        workspace_id: workspaceId,
        stage_id: firstStage.id,
        name: row.name.trim(),
        phone,
        email: row.email?.trim() || null,
        company: row.company?.trim() || null,
        source: row.source?.trim() || 'importação',
      })
    }

    // Inserir em lote
    if (toInsert.length > 0) {
      const { error } = await supabase.from('leads').insert(toInsert)
      if (error) throw new AppError('Erro ao inserir leads', 500)
      result.imported = toInsert.length
    }

    return result
  }
}
```

**Configuração Multer:**
```typescript
import multer from 'multer'

export const upload = multer({
  storage: multer.memoryStorage(),
  limits: { fileSize: 5 * 1024 * 1024 }, // 5MB
  fileFilter: (req, file, cb) => {
    if (file.mimetype === 'text/csv' || file.originalname.endsWith('.csv')) {
      cb(null, true)
    } else {
      cb(new Error('Apenas arquivos CSV são permitidos'))
    }
  },
})
```
