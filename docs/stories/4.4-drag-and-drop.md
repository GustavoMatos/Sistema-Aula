## Fluxo de Drag and Drop

```
1. Usuário inicia arraste do card
   └─ Card "levanta" com sombra
   └─ Cursor muda para grab

2. Usuário move sobre as colunas
   └─ Coluna destino destaca borda
   └─ Outras colunas ficam opacas

3. Usuário solta na coluna destino
   └─ Card move instantaneamente (otimista)
   └─ API chamada em background
   └─ Se sucesso: nada a fazer
   └─ Se erro: card volta + toast de erro
```

**KanbanBoard.tsx com DnD:**
```typescript
import {
  DndContext,
  DragOverlay,
  closestCorners,
  PointerSensor,
  TouchSensor,
  useSensor,
  useSensors,
} from '@dnd-kit/core'

export function KanbanBoard({ stages }: Props) {
  const [activeId, setActiveId] = useState<string | null>(null)
  const moveLead = useMoveLead()
  const queryClient = useQueryClient()

  const sensors = useSensors(
    useSensor(PointerSensor, {
      activationConstraint: { distance: 8 },
    }),
    useSensor(TouchSensor, {
      activationConstraint: { delay: 250, tolerance: 5 },
    })
  )

  const handleDragStart = (event: DragStartEvent) => {
    setActiveId(event.active.id as string)
  }

  const handleDragEnd = async (event: DragEndEvent) => {
    const { active, over } = event
    setActiveId(null)

    if (!over) return

    const leadId = active.id as string
    const newStageId = over.id as string

    // Find current stage
    const currentStage = stages.find(s =>
      s.leads.some(l => l.id === leadId)
    )

    if (currentStage?.id === newStageId) return

    // Optimistic update
    queryClient.setQueryData(['kanban'], (old: any) => {
      return moveLeadOptimistic(old, leadId, newStageId)
    })

    try {
      await moveLead.mutateAsync({ leadId, stageId: newStageId })
    } catch (error) {
      // Rollback
      queryClient.invalidateQueries({ queryKey: ['kanban'] })
      toast.error('Erro ao mover lead')
    }
  }

  const activeLead = activeId
    ? stages.flatMap(s => s.leads).find(l => l.id === activeId)
    : null

  return (
    <DndContext
      sensors={sensors}
      collisionDetection={closestCorners}
      onDragStart={handleDragStart}
      onDragEnd={handleDragEnd}
    >
      <div className="h-full overflow-x-auto">
        <div className="flex gap-4 p-4 h-full min-w-max">
          {stages.map(stage => (
            <DroppableColumn key={stage.id} stage={stage} />
          ))}
        </div>
      </div>

      <DragOverlay>
        {activeLead && (
          <KanbanCard lead={activeLead} isDragging />
        )}
      </DragOverlay>
    </DndContext>
  )
}
```

```typescript
import { useDroppable } from '@dnd-kit/core'

export function DroppableColumn({ stage }: Props) {
  const { setNodeRef, isOver } = useDroppable({
    id: stage.id,
  })

  return (
    <div
      ref={setNodeRef}
      className={cn(
        'flex flex-col w-80 shrink-0 rounded-lg transition-colors',
        isOver ? 'bg-primary/10 ring-2 ring-primary' : 'bg-muted/50'
      )}
    >
      {/* Header */}
      <div className="p-3 border-b">
        <div className="flex items-center gap-2">
          <div
            className="w-3 h-3 rounded-full"
            style={{ backgroundColor: stage.color }}
          />
          <h3 className="font-medium">{stage.name}</h3>
          <Badge variant="secondary">{stage.total_leads}</Badge>
        </div>
      </div>

      {/* Cards */}
      <div className="flex-1 overflow-y-auto p-2 space-y-2">
        {stage.leads.map(lead => (
          <DraggableCard key={lead.id} lead={lead} />
        ))}
      </div>
    </div>
  )
}
```

```typescript
import { useDraggable } from '@dnd-kit/core'

export function DraggableCard({ lead }: Props) {
  const { attributes, listeners, setNodeRef, transform, isDragging } = useDraggable({
    id: lead.id,
  })

  const style = {
    transform: transform
      ? `translate(${transform.x}px, ${transform.y}px)`
      : undefined,
  }

  return (
    <div
      ref={setNodeRef}
      style={style}
      {...listeners}
      {...attributes}
      className={cn(isDragging && 'opacity-50')}
    >
      <KanbanCard lead={lead} />
    </div>
  )
}
```

**Função de atualização otimista:**
```typescript
function moveLeadOptimistic(data: KanbanData, leadId: string, newStageId: string) {
  const stages = data.stages.map(stage => {
    // Remove from current stage
    const leads = stage.leads.filter(l => l.id !== leadId)

    // Add to new stage
    if (stage.id === newStageId) {
      const leadToMove = data.stages
        .flatMap(s => s.leads)
        .find(l => l.id === leadId)

      if (leadToMove) {
        leads.unshift(leadToMove)
      }
    }

    return {
      ...stage,
      leads,
      total_leads: leads.length,
    }
  })

  return { stages }
}
```
