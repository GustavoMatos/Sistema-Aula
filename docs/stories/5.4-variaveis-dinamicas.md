## Implementação

```typescript
const variableMap = {
  '{{nome}}': (lead: Lead) => lead.name || 'Cliente',
  '{{primeiro_nome}}': (lead: Lead) => lead.name?.split(' ')[0] || 'Cliente',
  '{{empresa}}': (lead: Lead) => lead.company || 'sua empresa',
  '{{telefone}}': (lead: Lead) => formatPhone(lead.phone),
  '{{email}}': (lead: Lead) => lead.email || '',
}

export function processVariables(text: string, lead: Lead): string {
  let result = text

  for (const [variable, getValue] of Object.entries(variableMap)) {
    result = result.replaceAll(variable, getValue(lead))
  }

  return result
}
```
