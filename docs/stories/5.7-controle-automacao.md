## Notas Técnicas

**AutomationToggle.tsx:**
```typescript
export function AutomationToggle({ lead }: { lead: Lead }) {
  const toggleAutomation = useToggleAutomation()

  return (
    <div className="flex items-center justify-between p-4 border rounded-lg">
      <div>
        <p className="font-medium">Automação de Follow-up</p>
        <p className="text-sm text-muted-foreground">
          {lead.automation_paused
            ? 'Mensagens automáticas pausadas'
            : 'Mensagens automáticas ativas'}
        </p>
      </div>
      <Switch
        checked={!lead.automation_paused}
        onCheckedChange={() => toggleAutomation.mutate(lead.id)}
      />
    </div>
  )
}
```
