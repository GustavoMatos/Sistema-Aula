## Notas Técnicas

**Hook useLeads.ts:**
```typescript
interface LeadFilters {
  stage_id?: string
  tags?: string[]
  source?: string
  search?: string
  sort?: string
  order?: 'asc' | 'desc'
  page?: number
  limit?: number
}

export function useLeads(filters: LeadFilters = {}) {
  return useQuery({
    queryKey: ['leads', filters],
    queryFn: () => api.leads.list(filters),
    keepPreviousData: true,
  })
}

export function useBulkMoveStage() {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: ({ leadIds, stageId }: { leadIds: string[]; stageId: string }) =>
      Promise.all(leadIds.map(id => api.leads.moveStage(id, stageId))),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['leads'] })
      toast.success('Leads movidos com sucesso')
    },
  })
}
```

**LeadsTable.tsx:**
```typescript
export function LeadsTable({ leads, selected, onSelect, onSelectAll }: Props) {
  return (
    <Table>
      <TableHeader>
        <TableRow>
          <TableHead className="w-12">
            <Checkbox
              checked={selected.length === leads.length}
              onCheckedChange={onSelectAll}
            />
          </TableHead>
          <TableHead>Nome</TableHead>
          <TableHead>Telefone</TableHead>
          <TableHead>Estágio</TableHead>
          <TableHead>Origem</TableHead>
          <TableHead>Última Interação</TableHead>
        </TableRow>
      </TableHeader>
      <TableBody>
        {leads.map(lead => (
          <LeadRow
            key={lead.id}
            lead={lead}
            selected={selected.includes(lead.id)}
            onSelect={() => onSelect(lead.id)}
          />
        ))}
      </TableBody>
    </Table>
  )
}
```

**BulkActionsBar.tsx:**
```typescript
export function BulkActionsBar({ selectedIds, onClear }: Props) {
  const moveStage = useBulkMoveStage()
  const { data: stages } = useStages()

  if (selectedIds.length === 0) return null

  return (
    <div className="bg-muted p-3 rounded-lg flex items-center gap-4">
      <span className="text-sm">
        {selectedIds.length} selecionado(s)
      </span>

      <Select onValueChange={(stageId) => moveStage.mutate({ leadIds: selectedIds, stageId })}>
        <SelectTrigger className="w-40">
          <SelectValue placeholder="Mover Estágio" />
        </SelectTrigger>
        <SelectContent>
          {stages?.map(stage => (
            <SelectItem key={stage.id} value={stage.id}>
              {stage.name}
            </SelectItem>
          ))}
        </SelectContent>
      </Select>

      <Button variant="ghost" size="sm" onClick={onClear}>
        Limpar seleção
      </Button>
    </div>
  )
}
```
