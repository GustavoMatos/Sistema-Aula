## Notas Técnicas

**Hook useWhatsApp.ts:**
```typescript
export function useWhatsAppInstances() {
  return useQuery({
    queryKey: ['whatsapp-instances'],
    queryFn: () => api.whatsapp.listInstances(),
  })
}

export function useCreateInstance() {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: (name: string) => api.whatsapp.createInstance(name),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['whatsapp-instances'] })
    },
  })
}

export function useDeleteInstance() {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: (id: string) => api.whatsapp.deleteInstance(id),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['whatsapp-instances'] })
    },
  })
}
```

**StatusBadge Component:**
```typescript
const statusConfig = {
  connected: { color: 'bg-green-500', label: 'Conectado' },
  disconnected: { color: 'bg-red-500', label: 'Desconectado' },
  connecting: { color: 'bg-yellow-500', label: 'Conectando' },
}

export function StatusBadge({ status }: { status: InstanceStatus }) {
  const config = statusConfig[status]
  return (
    <span className="flex items-center gap-2">
      <span className={`w-2 h-2 rounded-full ${config.color}`} />
      {config.label}
    </span>
  )
}
```
