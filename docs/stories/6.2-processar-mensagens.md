## Payload MESSAGES_UPSERT

```json
{
  "event": "messages.upsert",
  "instance": "minha-instancia",
  "data": {
    "key": {
      "remoteJid": "5511999999999@s.whatsapp.net",
      "fromMe": false,
      "id": "3EB0B430A5..."
    },
    "pushName": "Nome do Contato",
    "message": {
      "conversation": "Texto da mensagem"
    },
    "messageType": "conversation",
    "messageTimestamp": 1717689097
  }
}
```

## Implementação

```typescript
// message-handler.ts
export async function handleMessageUpsert(instance: string, data: any) {
  const { key, pushName, message, messageTimestamp } = data

  // Extrair número
  const phone = extractPhoneNumber(key.remoteJid)
  const text = message?.conversation || message?.extendedTextMessage?.text || ''

  // Buscar lead
  let lead = await findLeadByPhone(phone)

  // Se não existe, criar
  if (!lead) {
    lead = await createLeadFromWhatsApp(phone, pushName)
  }

  // Salvar mensagem
  await saveMessage({
    lead_id: lead.id,
    direction: 'inbound',
    content: text,
    message_id: key.id,
    sent_at: new Date(messageTimestamp * 1000)
  })

  // Atualizar last_contact_at
  await updateLeadLastContact(lead.id)

  // Pausar automação se configurado
  if (lead.automation_paused === false) {
    await pauseAutomationTemporarily(lead.id)
  }

  // Registrar histórico
  await createLeadHistory(lead.id, 'message_received')
}
```

## Extração do Número

```typescript
function extractPhoneNumber(remoteJid: string): string {
  // "5511999999999@s.whatsapp.net" -> "5511999999999"
  return remoteJid.split('@')[0]
}
```
