## Notas Técnicas

**MessageComposer.tsx:**
```typescript
export function MessageComposer({ leadId }: { leadId: string }) {
  const [message, setMessage] = useState('')
  const [image, setImage] = useState<File | null>(null)
  const [imagePreview, setImagePreview] = useState<string | null>(null)

  const sendMessage = useSendMessage()
  const queryClient = useQueryClient()

  const handleImageSelect = (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0]
    if (file) {
      setImage(file)
      setImagePreview(URL.createObjectURL(file))
    }
  }

  const handleSend = async () => {
    if (!message.trim() && !image) return

    try {
      if (image) {
        // Upload image first (to storage or use base64)
        const mediaUrl = await uploadImage(image)
        await sendMessage.mutateAsync({
          lead_id: leadId,
          type: 'image',
          media_url: mediaUrl,
          caption: message,
        })
      } else {
        await sendMessage.mutateAsync({
          lead_id: leadId,
          type: 'text',
          content: message,
        })
      }

      // Clear form
      setMessage('')
      setImage(null)
      setImagePreview(null)

      // Refresh messages
      queryClient.invalidateQueries({ queryKey: ['messages', leadId] })
    } catch (error) {
      toast.error('Erro ao enviar mensagem')
    }
  }

  return (
    <div className="border-t p-4">
      {imagePreview && (
        <ImagePreview
          src={imagePreview}
          onRemove={() => {
            setImage(null)
            setImagePreview(null)
          }}
        />
      )}

      <div className="flex items-center gap-2">
        <input
          type="file"
          accept="image/*"
          className="hidden"
          id="image-input"
          onChange={handleImageSelect}
        />
        <Button
          variant="ghost"
          size="icon"
          onClick={() => document.getElementById('image-input')?.click()}
        >
          <ImageIcon className="w-5 h-5" />
        </Button>

        <Input
          placeholder="Digite sua mensagem..."
          value={message}
          onChange={(e) => setMessage(e.target.value)}
          onKeyDown={(e) => e.key === 'Enter' && handleSend()}
        />

        <Button
          onClick={handleSend}
          disabled={sendMessage.isPending || (!message.trim() && !image)}
        >
          {sendMessage.isPending ? (
            <Loader2 className="w-4 h-4 animate-spin" />
          ) : (
            <Send className="w-4 h-4" />
          )}
        </Button>
      </div>
    </div>
  )
}
```

**useSendMessage Hook:**
```typescript
export function useSendMessage() {
  return useMutation({
    mutationFn: (data: SendMessageDTO) => api.messages.send(data),
    onError: (error) => {
      if (error.message.includes('No WhatsApp instance')) {
        toast.error('Nenhuma instância WhatsApp conectada')
      } else {
        toast.error('Erro ao enviar mensagem')
      }
    },
  })
}
```
