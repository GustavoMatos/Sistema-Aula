## Componente RecentLeadsList

```typescript
export function RecentLeadsList() {
  const { leads, isLoading } = useRecentLeads()
  const navigate = useNavigate()

  if (isLoading) {
    return (
      <Card>
        <CardHeader>
          <CardTitle>Leads Recentes</CardTitle>
        </CardHeader>
        <CardContent>
          {[1, 2, 3, 4, 5].map((i) => (
            <Skeleton key={i} className="h-12 mb-2" />
          ))}
        </CardContent>
      </Card>
    )
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center gap-2">
          <Users className="w-5 h-5" />
          Leads Recentes
        </CardTitle>
      </CardHeader>
      <CardContent className="space-y-1">
        {leads.length === 0 ? (
          <p className="text-muted-foreground text-sm py-4">
            Nenhum lead cadastrado ainda
          </p>
        ) : (
          leads.map((lead) => (
            <RecentLeadItem
              key={lead.id}
              lead={lead}
              onClick={() => navigate(`/leads/${lead.id}`)}
            />
          ))
        )}
      </CardContent>
      <CardFooter>
        <Link
          to="/leads"
          className="text-sm text-blue-600 hover:underline"
        >
          Ver todos os leads →
        </Link>
      </CardFooter>
    </Card>
  )
}
```

## Componente RecentLeadItem

```typescript
interface RecentLeadItemProps {
  lead: Lead
  onClick: () => void
}

export function RecentLeadItem({ lead, onClick }: RecentLeadItemProps) {
  return (
    <div
      className="flex items-center justify-between p-2 rounded-lg hover:bg-muted/50 cursor-pointer transition-colors"
      onClick={onClick}
    >
      <div className="flex items-center gap-3">
        {/* Avatar */}
        <Avatar className="w-8 h-8">
          <AvatarFallback className="text-xs">
            {getInitials(lead.name)}
          </AvatarFallback>
        </Avatar>

        <div>
          <p className="font-medium text-sm">{lead.name}</p>
          <div className="flex items-center gap-2">
            <Badge
              variant="outline"
              className="text-xs"
              style={{
                borderColor: lead.stage?.color,
                color: lead.stage?.color
              }}
            >
              {lead.stage?.name}
            </Badge>
          </div>
        </div>
      </div>

      <span className="text-xs text-muted-foreground">
        {formatRelativeTime(lead.created_at)}
      </span>
    </div>
  )
}

function getInitials(name: string): string {
  return name
    .split(' ')
    .map(n => n[0])
    .join('')
    .toUpperCase()
    .slice(0, 2)
}
```

## Hook useRecentLeads

```typescript
export function useRecentLeads(limit = 5) {
  return useQuery({
    queryKey: ['leads', 'recent', limit],
    queryFn: async () => {
      const { data, error } = await supabase
        .from('leads')
        .select(`
          id,
          name,
          created_at,
          stage:kanban_stages(name, color)
        `)
        .order('created_at', { ascending: false })
        .limit(limit)

      if (error) throw error
      return data
    }
  })
}
```

## Utilitário formatRelativeTime

```typescript
export function formatRelativeTime(date: string | Date): string {
  const now = new Date()
  const then = new Date(date)
  const diffMs = now.getTime() - then.getTime()
  const diffMins = Math.floor(diffMs / 60000)
  const diffHours = Math.floor(diffMs / 3600000)
  const diffDays = Math.floor(diffMs / 86400000)

  if (diffMins < 1) return 'agora'
  if (diffMins < 60) return `${diffMins}min`
  if (diffHours < 24) return `${diffHours}h`
  if (diffDays < 7) return `${diffDays}d`

  return then.toLocaleDateString('pt-BR', {
    day: 'numeric',
    month: 'short'
  })
}
```
