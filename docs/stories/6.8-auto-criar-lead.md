## Implementação

```typescript
// lead.service.ts
export async function createLeadFromWhatsApp(
  phone: string,
  pushName: string | null,
  workspaceId: string
): Promise<Lead> {
  // Buscar primeiro estágio
  const { data: firstStage } = await supabase
    .from('kanban_stages')
    .select('id')
    .eq('workspace_id', workspaceId)
    .order('position', { ascending: true })
    .limit(1)
    .single()

  if (!firstStage) {
    throw new Error('No stages found for workspace')
  }

  // Criar lead
  const { data: lead, error } = await supabase
    .from('leads')
    .insert({
      workspace_id: workspaceId,
      stage_id: firstStage.id,
      name: pushName || 'Lead WhatsApp',
      phone: phone,
      source: 'WhatsApp',
      status: 'active',
      automation_paused: false
    })
    .select()
    .single()

  if (error) throw error

  // Registrar histórico
  await supabase
    .from('lead_history')
    .insert({
      lead_id: lead.id,
      action: 'lead_created',
      changes: {
        source: 'WhatsApp',
        auto_created: true
      }
    })

  console.log(`[Lead] Auto-created: ${lead.name} (${phone})`)

  return lead
}
```

## Buscar Lead por Telefone

```typescript
export async function findLeadByPhone(
  phone: string,
  workspaceId: string
): Promise<Lead | null> {
  // Normalizar número
  const normalizedPhone = normalizePhone(phone)

  const { data } = await supabase
    .from('leads')
    .select('*')
    .eq('workspace_id', workspaceId)
    .or(`phone.eq.${normalizedPhone},phone.eq.${phone}`)
    .limit(1)
    .single()

  return data
}

function normalizePhone(phone: string): string {
  // Remove caracteres não numéricos
  return phone.replace(/\D/g, '')
}
```
