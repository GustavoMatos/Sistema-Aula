## Notas Técnicas

**LeadsService:**
```typescript
export class LeadsService {
  async list(workspaceId: string, filters: LeadFilters) {
    let query = supabase
      .from('leads')
      .select('*, kanban_stages(name, color)', { count: 'exact' })
      .eq('workspace_id', workspaceId)

    if (filters.stage_id) {
      query = query.eq('stage_id', filters.stage_id)
    }

    if (filters.search) {
      query = query.or(
        `name.ilike.%${filters.search}%,phone.ilike.%${filters.search}%`
      )
    }

    if (filters.tags?.length) {
      query = query.overlaps('tags', filters.tags)
    }

    if (filters.source) {
      query = query.eq('source', filters.source)
    }

    // Ordenação
    const order = filters.order === 'asc' ? true : false
    query = query.order(filters.sort || 'created_at', { ascending: order })

    // Paginação
    const from = (filters.page - 1) * filters.limit
    const to = from + filters.limit - 1
    query = query.range(from, to)

    const { data, error, count } = await query

    return {
      leads: data,
      total: count,
      page: filters.page,
      limit: filters.limit,
      totalPages: Math.ceil((count || 0) / filters.limit),
    }
  }
}
```
