## Payload CONNECTION_UPDATE

```json
{
  "event": "connection.update",
  "instance": "minha-instancia",
  "data": {
    "instance": "minha-instancia",
    "state": "open"
  }
}
```

---

## Mapeamento de Estados

| Evolution API | Sistema |
|---------------|---------|
| `open` | `connected` |
| `close` | `disconnected` |
| `connecting` | `connecting` |

---

## Implementação

```typescript
// connection-handler.ts
export async function handleConnectionUpdate(instance: string, data: any) {
  const { state } = data

  // Mapear estado
  const statusMap: Record<string, string> = {
    'open': 'connected',
    'close': 'disconnected',
    'connecting': 'connecting'
  }

  const status = statusMap[state] || 'disconnected'

  // Buscar instância pelo nome
  const whatsappInstance = await findInstanceByName(instance)

  if (!whatsappInstance) {
    console.warn(`[Connection] Instance not found: ${instance}`)
    return
  }

  // Atualizar status
  await updateInstanceStatus(whatsappInstance.id, status)

  // Se conectou, limpar QR code
  if (status === 'connected') {
    await clearInstanceQRCode(whatsappInstance.id)
  }

  console.log(`[Connection] ${instance} -> ${status}`)
}
```
