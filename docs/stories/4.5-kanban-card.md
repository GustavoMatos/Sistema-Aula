**KanbanCard.tsx:**
```typescript
export function KanbanCard({ lead, isDragging }: Props) {
  const navigate = useNavigate()
  const [showActions, setShowActions] = useState(false)

  return (
    <Card
      className={cn(
        'relative cursor-grab active:cursor-grabbing',
        'hover:shadow-md transition-all',
        isDragging && 'shadow-lg scale-105 rotate-2'
      )}
      onMouseEnter={() => setShowActions(true)}
      onMouseLeave={() => setShowActions(false)}
    >
      <CardContent className="p-3">
        {/* Header com nome e menu */}
        <div className="flex items-start justify-between">
          <h4 className="font-medium truncate max-w-[180px]" title={lead.name}>
            {lead.name}
          </h4>
          <DropdownMenu>
            <DropdownMenuTrigger asChild>
              <Button variant="ghost" size="icon" className="h-6 w-6">
                <MoreVertical className="h-4 w-4" />
              </Button>
            </DropdownMenuTrigger>
            <DropdownMenuContent>
              <DropdownMenuItem onClick={() => navigate(`/leads/${lead.id}`)}>
                <Eye className="w-4 h-4 mr-2" />
                Ver detalhes
              </DropdownMenuItem>
              <DropdownMenuItem>
                <MessageSquare className="w-4 h-4 mr-2" />
                Enviar mensagem
              </DropdownMenuItem>
              <DropdownMenuSeparator />
              <DropdownMenuItem className="text-destructive">
                <Trash className="w-4 h-4 mr-2" />
                Excluir
              </DropdownMenuItem>
            </DropdownMenuContent>
          </DropdownMenu>
        </div>

        {/* Telefone */}
        <p className="text-sm text-muted-foreground">
          {formatPhone(lead.phone)}
        </p>

        {/* Tags */}
        <div className="flex gap-1 mt-2 flex-wrap">
          {lead.tags.slice(0, 2).map(tag => (
            <Badge key={tag} variant="outline" className="text-xs px-1.5 py-0">
              {tag}
            </Badge>
          ))}
          {lead.tags.length > 2 && (
            <Badge variant="secondary" className="text-xs px-1.5 py-0">
              +{lead.tags.length - 2}
            </Badge>
          )}
        </div>

        {/* Footer: tempo e indicadores */}
        <div className="flex items-center justify-between mt-2 text-xs text-muted-foreground">
          <span>{formatDaysInStage(lead.days_in_stage)}</span>
          <div className="flex items-center gap-1">
            {lead.automation_paused && (
              <Tooltip>
                <TooltipTrigger>
                  <PauseCircle className="w-3.5 h-3.5 text-yellow-500" />
                </TooltipTrigger>
                <TooltipContent>Automação pausada</TooltipContent>
              </Tooltip>
            )}
            {lead.potential_value && (
              <span className="text-green-600 font-medium">
                {formatCurrency(lead.potential_value)}
              </span>
            )}
          </div>
        </div>

        {/* Quick Actions (hover) */}
        {showActions && (
          <KanbanCardActions lead={lead} />
        )}
      </CardContent>
    </Card>
  )
}
```

**Funções de formatação:**
```typescript
function formatPhone(phone: string): string {
  const digits = phone.replace(/\D/g, '')
  if (digits.length === 11) {
    return `(${digits.slice(0, 2)}) ${digits.slice(2, 7)}-${digits.slice(7)}`
  }
  if (digits.length === 10) {
    return `(${digits.slice(0, 2)}) ${digits.slice(2, 6)}-${digits.slice(6)}`
  }
  return phone
}

function formatDaysInStage(days: number): string {
  if (days === 0) return 'hoje'
  if (days === 1) return 'há 1 dia'
  return `há ${days} dias`
}

function formatCurrency(value: number): string {
  return new Intl.NumberFormat('pt-BR', {
    style: 'currency',
    currency: 'BRL',
    minimumFractionDigits: 0,
  }).format(value)
}
```
