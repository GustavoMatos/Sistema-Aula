## Componente WhatsAppStatusCard

```typescript
export function WhatsAppStatusCard() {
  const { instances, isLoading } = useInstances()
  const reconnect = useReconnectInstance()

  if (isLoading) {
    return (
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            <Phone className="w-5 h-5" />
            WhatsApp
          </CardTitle>
        </CardHeader>
        <CardContent>
          <Skeleton className="h-20 w-full" />
        </CardContent>
      </Card>
    )
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center gap-2">
          <Phone className="w-5 h-5" />
          WhatsApp
        </CardTitle>
      </CardHeader>
      <CardContent className="space-y-3">
        {instances.length === 0 ? (
          <p className="text-muted-foreground text-sm">
            Nenhuma instância configurada
          </p>
        ) : (
          instances.map((instance) => (
            <InstanceStatus
              key={instance.id}
              instance={instance}
              onReconnect={() => reconnect.mutate(instance.id)}
            />
          ))
        )}
      </CardContent>
      <CardFooter>
        <Link
          to="/settings/whatsapp"
          className="text-sm text-blue-600 hover:underline"
        >
          Gerenciar instâncias →
        </Link>
      </CardFooter>
    </Card>
  )
}
```

## Componente InstanceStatus

```typescript
interface InstanceStatusProps {
  instance: WhatsAppInstance
  onReconnect: () => void
}

export function InstanceStatus({ instance, onReconnect }: InstanceStatusProps) {
  const isConnected = instance.status === 'connected'
  const isConnecting = instance.status === 'connecting'

  return (
    <div className="flex items-center justify-between p-2 rounded-lg bg-muted/50">
      <div className="flex items-center gap-3">
        {/* Status Indicator */}
        <div className={cn(
          'w-2 h-2 rounded-full',
          isConnected && 'bg-green-500',
          isConnecting && 'bg-yellow-500 animate-pulse',
          !isConnected && !isConnecting && 'bg-red-500'
        )} />

        <div>
          <p className="font-medium text-sm">{instance.name}</p>
          <p className="text-xs text-muted-foreground">
            {isConnected && 'Conectado'}
            {isConnecting && 'Conectando...'}
            {!isConnected && !isConnecting && (
              instance.last_connected_at
                ? `Desconectado há ${formatRelativeTime(instance.last_connected_at)}`
                : 'Nunca conectado'
            )}
          </p>
        </div>
      </div>

      {/* Actions */}
      {!isConnected && !isConnecting && (
        <Button
          variant="ghost"
          size="sm"
          onClick={onReconnect}
        >
          <RefreshCw className="w-4 h-4" />
        </Button>
      )}
    </div>
  )
}
```

## Hook useInstances

```typescript
export function useInstances() {
  const queryClient = useQueryClient()

  const query = useQuery({
    queryKey: ['whatsapp-instances'],
    queryFn: () => api.get('/whatsapp/instances')
  })

  // Realtime subscription
  useEffect(() => {
    const channel = supabase
      .channel('instances-status')
      .on('postgres_changes', {
        event: 'UPDATE',
        schema: 'public',
        table: 'whatsapp_instances'
      }, () => {
        queryClient.invalidateQueries({ queryKey: ['whatsapp-instances'] })
      })
      .subscribe()

    return () => { channel.unsubscribe() }
  }, [queryClient])

  return {
    instances: query.data || [],
    isLoading: query.isLoading
  }
}

export function useReconnectInstance() {
  return useMutation({
    mutationFn: (instanceId: string) =>
      api.post(`/whatsapp/instances/${instanceId}/reconnect`)
  })
}
```
