## Endpoints

### POST /api/stages
```json
// Request
{
  "name": "Em Análise",
  "color": "#f59e0b",
  "is_final": false
}

// Response 201
{
  "id": "uuid",
  "name": "Em Análise",
  "color": "#f59e0b",
  "position": 7,
  "is_final": false
}
```

### GET /api/stages
```json
// Response 200
{
  "stages": [
    { "id": "uuid", "name": "Novo Lead", "color": "#6366f1", "position": 0, "is_final": false },
    { "id": "uuid", "name": "Primeiro Contato", "color": "#8b5cf6", "position": 1, "is_final": false },
    ...
  ]
}
```

### PATCH /api/stages/reorder
```json
// Request
{
  "order": ["uuid-1", "uuid-2", "uuid-3", ...]

// Response 200
{ "success": true }
```

```json
// Response 200 (success)
{ "success": true }

// Response 400 (has leads)
{
  "error": "Cannot delete stage with leads",
  "leads_count": 15
}
```

## Notas Técnicas

**StagesService:**
```typescript
export class StagesService {
  async list(workspaceId: string): Promise<Stage[]> {
    const { data } = await supabase
      .from('kanban_stages')
      .select('*')
      .eq('workspace_id', workspaceId)
      .order('position', { ascending: true })

    return data || []
  }

  async create(workspaceId: string, data: CreateStageDTO): Promise<Stage> {
    // Get max position
    const { data: stages } = await supabase
      .from('kanban_stages')
      .select('position')
      .eq('workspace_id', workspaceId)
      .order('position', { ascending: false })
      .limit(1)

    const nextPosition = (stages?.[0]?.position ?? -1) + 1

    const { data: stage, error } = await supabase
      .from('kanban_stages')
      .insert({
        workspace_id: workspaceId,
        name: data.name,
        color: data.color || '#6366f1',
        position: nextPosition,
        is_final: data.is_final || false,
      })
      .select()
      .single()

    if (error) throw new AppError('Failed to create stage', 500)
    return stage
  }

  async delete(id: string): Promise<void> {
    // Check for leads
    const { count } = await supabase
      .from('leads')
      .select('*', { count: 'exact', head: true })
      .eq('stage_id', id)

    if (count && count > 0) {
      throw new AppError('Cannot delete stage with leads', 400, { leads_count: count })
    }

    await supabase.from('kanban_stages').delete().eq('id', id)
  }

  async reorder(workspaceId: string, order: string[]): Promise<void> {
    const updates = order.map((id, index) => ({
      id,
      position: index,
    }))

    for (const update of updates) {
      await supabase
        .from('kanban_stages')
        .update({ position: update.position })
        .eq('id', update.id)
        .eq('workspace_id', workspaceId)
    }
  }
}
```
